create Class Sprawa(
	ID_Sprawa int,
	Numer int,
	Data String,
	Sala string,
	Zatrudnila References set (Pracownik) Cardinality (0,-1) Inverse Pracownik.Pracuje,
	Posiada References set (UczestnikSprawy) Cardinality (0,-1) Inverse UczestnikSprawy.Uczestniczy
